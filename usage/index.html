import web

resources = (
  r'/(.*)', 'Hello'
)

class Hello:
  """
  say Hello
  
  """
  
  def GET(self, name):
    if not name: 
        name = 'World'
    return 'Hello ' + name + '!'

if __name__ == '__main__':
  webapp.run()
